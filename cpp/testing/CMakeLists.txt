cmake_minimum_required(VERSION 3.15)
set(CMAKE_CXX_STANDARD 11)
project(SVR)
set(BINARY ${CMAKE_PROJECT_NAME}_tests)
set(PROFILING_BINARY ${CMAKE_PROJECT_NAME}_profiling)
set(SOURCE_FILES ../spherical_volume_rendering_util.h ../spherical_volume_rendering_util.cpp spherical_volume_rendering_test.cpp)
set(PROFILER_SOURCE_FILES ../spherical_volume_rendering_util.h ../spherical_volume_rendering_util.cpp profiling.cpp)

add_subdirectory(googletest)
include_directories(${../cpp})
include_directories(${gtest_SOURCE_DIR}/include ${gtest_SOURCE_DIR})
include_directories(${gmock_SOURCE_DIR}/include ${gmock_SOURCE_DIR})

add_executable(${PROFILING_BINARY} ${PROFILER_SOURCE_FILES})

target_link_libraries(${BINARY} gtest gtest_main gmock gmock_main)
